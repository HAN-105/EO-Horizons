namespace = uu_game_start

## ENSURES THAT ALL STARTER PLANETS HAVE PROPER PLANET CLASS
## THIS IS FOR STATIC GALAXIES AS THE GENERATION IS BUGGY...
event = {
	id = uu_game_start.2
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_country = {
			limit = {
				is_country_type = default
			}
			capital_scope = {
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_continental_preference }
							NOT = { is_planet_class = pc_continental }
						}
					}
					change_pc = pc_continental
				}
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_tropical_preference }
							NOT = { is_planet_class = pc_tropical }
						}
					}
					change_pc = pc_tropical
				}
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_desert_preference }
							NOT = { is_planet_class = pc_desert }
						}
					}
					change_pc = pc_desert
				}
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_ocean_preference }
							NOT = { is_planet_class = pc_ocean }
						}
					}
					change_pc = pc_ocean
				}
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_arctic_preference }
							NOT = { is_planet_class = pc_arctic }
						}
					}
					change_pc = pc_arctic
				}
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_tundra_preference }
							NOT = { is_planet_class = pc_tundra }
						}
					}
					change_pc = pc_tundra
				}
				if = {
					limit = {
						AND = {
							owner = { has_trait = trait_pc_arid_preference }
							NOT = { is_planet_class = pc_arid }
						}
					}
					change_pc = pc_arid
				}
				
				owner = {
					species = {
						set_species_homeworld = PREVPREV
					}
				}
			}
		}
	}
}

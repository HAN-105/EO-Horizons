
#### Takmahl Colony Category
anomaly_category = {
	key = "takmahl_colony_category"
	desc = "takmahl_colony_category_desc"
	picture = GFX_evt_ship_in_orbit
	level = 1
	risk = 20
	
	spawn_chance = {		
		modifier = {
			add = 200
			OR = {
				is_planet_class = pc_arctic
				is_planet_class = pc_continental
				is_planet_class = pc_tropical
				is_planet_class = pc_arid
				is_planet_class = pc_desert
				is_planet_class = pc_tundra
				is_planet_class = pc_ocean
			}
			solar_system = { has_star_flag = precursor_takmahl }
		}
	}
	
	on_spawn = {
		from = {
			owner = {
				if = {
					limit = { has_country_flag = precursors_found }
				}
				set_country_flag = precursors_found
			}
		}
	}
	
	on_success = {
	}

	on_fail = {
		ship_event = { id = anomaly_failure.33 }
	}

	on_critical_fail = {
		random_list = {
			16 = { ship_event = { id = anomaly_failure.400 } }
			16 = { ship_event = { id = anomaly_failure.401 } }
			16 = { ship_event = { id = anomaly_failure.402 } }
			16 = { ship_event = { id = anomaly_failure.403 } }
			16 = { ship_event = { id = anomaly_failure.404 } }
			16 = { ship_event = { id = anomaly_failure.405 } }
		}
	}
}

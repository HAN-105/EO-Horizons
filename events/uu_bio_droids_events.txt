namespace = bio_droids


pop_event = {
	id = bio_droids.1
	hide_window = yes
	
	trigger = {
		AND = {
			is_species = "TAKMAHL_BIO_DROID_1"
			is_growing = yes
		}
		NOT = { has_pop_flag = is_processed }
	}
	
	immediate = {
		set_pop_flag = is_processed
		planet = {
			random_pop = {
				limit = {
					NOT = { is_species = "TAKMAHL_BIO_DROID_1" }
				}
				kill_pop = yes
			}
		}
	}
}
